# Sahil help commands

```
# creating folder
mkdir myfolder

#create two folder: my and folder
mkdir my folder

# print some text in git bash(or simply bash)
echo Hello world.

# print current working directory
pwd

# print content of a file
cat filename

# print git history
git log 
# or
git log --one-line

# print files and folders of current directory
ls

# print all files INCLUDING HIDDEN FILES
ls -a
```
